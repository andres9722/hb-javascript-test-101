export function isPalindrome(str) {
    let pal = str.replace(/\s/g,'').toLowerCase()
    let cleanString = pal.replace(/[\|&;\$%@"<>\(\)\+,*.]/g, "")
    return cleanString === cleanString.split('').reverse().join('')
}
